# SC-017.1 — Исследование: продажи автомобилей в России 2024-2025
# ВЕРСИЯ 2 — верифицированные URL, без шаблонной генерации

## Цель

Полный цикл исследования авторынка России. В этой версии URL-список
составлен только из проверенных источников — либо найденных через web_search,
либо заведомо открытых главных страниц. Никаких составных URL по шаблону.

---

## Урок из SC-017.1 v1

В первой версии 81.5% URL провалилось. Три главные причины:

**Несуществующие URL (~35%)** — конкретные ссылки вида
`autostat.ru/press-releases/61576/` составлялись по шаблону угадыванием ID.
На большинстве сайтов ID случайные или URL содержит slug которого не угадать.

**Пейволы (~20%)** — `kommersant.ru/doc/*`, `vedomosti.ru/*`,
`rbc.ru` отдельные статьи всегда закрыты. Открыты только разделы и главные.

**JS-SPA (~25%)** — `drom.ru/catalog/*`, `vc.ru`, `pikabu.ru`, `avito.ru/journal`
не рендерятся без браузера, возвращают пустую страницу.

**Правило на будущее**: URL для scout_index должны приходить из двух источников —
либо из результатов web_search (гарантированно существуют), либо из заведомо
открытых главных страниц и разделов сайтов. Никакого конструирования по шаблону.

## Шаг 2 — Поиск по 6 углам

```bash
python3 << 'SCRIPT'
import json, subprocess

def mcp_init():
    r = subprocess.run(
        ["curl", "-si", "-X", "POST", "http://localhost:8020/mcp",
         "-H", "Content-Type: application/json",
         "-H", "Accept: application/json, text/event-stream",
         "-d", '{"jsonrpc":"2.0","id":0,"method":"initialize","params":{"protocolVersion":"2024-11-05","capabilities":{},"clientInfo":{"name":"sc017","version":"1.0"}}}'],
        capture_output=True, text=True, timeout=30
    )
    for line in r.stdout.split("\n"):
        if line.lower().startswith("mcp-session-id:"):
            return line.split(":", 1)[1].strip()

sid = mcp_init()
with open("/tmp/sc017_session.txt") as f:
    session_id = f.read().strip()

queries = [
    "объём рынка продажи автомобилей статистика 2024 2025",
    "китайские автомобили доля рынка бренды Haval Chery Geely",
    "LADA АвтоВАЗ продажи модели Granta Vesta",
    "автокредиты ключевая ставка спрос покупки",
    "уход западных брендов параллельный импорт",
    "прогноз авторынок тренды 2025 2026",
]

for q in queries:
    payload = json.dumps({
        "jsonrpc":"2.0","id":1,"method":"tools/call",
        "params":{"name":"scout_search","arguments":{
            "session_id": session_id, "query": q, "top_k": 5
        }}
    })
    r = subprocess.run(
        ["curl","-s","-X","POST","http://localhost:8020/mcp",
         "-H","Content-Type: application/json",
         "-H","Accept: application/json, text/event-stream",
         "-H",f"Mcp-Session-Id: {sid}",
         "-d", payload],
        capture_output=True, text=True, timeout=60
    )
    body = r.stdout.strip()
    if body.startswith("data:"): body = body[5:].strip()
    try:
        d = json.loads(body)
        rs = d.get("result", {}).get("results", [])
        print(f"\n=== {q} ===")
        print(f"  results: {len(rs)}")
        for res in rs:
            print(f"  {res['similarity']:.3f} | {res['source_title'][:60]}")
    except Exception as e:
        print(f"ERR: {e}")
SCRIPT
```

## Шаг 3 — Brief и сохранение

```bash
python3 << 'SCRIPT'
import json, subprocess, os

def mcp_init():
    r = subprocess.run(
        ["curl","-si","-X","POST","http://localhost:8020/mcp",
         "-H","Content-Type: application/json",
         "-H","Accept: application/json, text/event-stream",
         "-d",'{"jsonrpc":"2.0","id":0,"method":"initialize","params":{"protocolVersion":"2024-11-05","capabilities":{},"clientInfo":{"name":"sc017","version":"1.0"}}}'],
        capture_output=True, text=True, timeout=30
    )
    for line in r.stdout.split("\n"):
        if line.lower().startswith("mcp-session-id:"):
            return line.split(":", 1)[1].strip()

sid = mcp_init()
with open("/tmp/sc017_session.txt") as f:
    session_id = f.read().strip()

payload = json.dumps({
    "jsonrpc":"2.0","id":1,"method":"tools/call",
    "params":{"name":"scout_brief","arguments":{
        "session_id": session_id,
        "query": (
            "Продуктовое исследование: рынок продаж автомобилей в России 2024-2025. "
            "Объём рынка и динамика, доминирование китайских брендов и ключевые игроки, "
            "позиции LADA и АвтоВАЗа, влияние ключевой ставки и автокредитов, "
            "уход западных производителей и параллельный импорт, "
            "тренды и прогноз на 2025-2026 год."
        ),
        "top_k": 15
    }}
})

r = subprocess.run(
    ["curl","-s","-X","POST","http://localhost:8020/mcp",
     "-H","Content-Type: application/json",
     "-H","Accept: application/json, text/event-stream",
     "-H",f"Mcp-Session-Id: {sid}",
     "-d", payload],
    capture_output=True, text=True, timeout=120
)
body = r.stdout.strip()
if body.startswith("data:"): body = body[5:].strip()

d = json.loads(body)
res = d.get("result", {})
brief   = res.get("brief", "")
tokens  = res.get("tokens_used")
sources = res.get("sources_used")
model   = res.get("model")

print(f"model: {model} | tokens: {tokens} | sources: {sources}")
print(f"\n{brief[:300]}...")

os.makedirs("/opt/scout/results", exist_ok=True)
with open("/opt/scout/results/SC-017.1_brief.md", "w") as f:
    f.write("# SC-017.1 — Авторынок России 2024-2025: продуктовый бриф\n\n")
    f.write(f"**Модель**: {model}  \n**Токены**: {tokens}  \n")
    f.write(f"**Источников**: {sources}  \n**Session**: {session_id}  \n\n---\n\n")
    f.write(brief)
print("Brief → /opt/scout/results/SC-017.1_brief.md")
SCRIPT
```
